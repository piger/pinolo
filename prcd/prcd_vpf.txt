TITLE   BESTEMMIA
;   PROGRAMMA ASSEMBLY PER FAR BESTEMMIARE IL PC - Versione 2.0
;   DA COMPILARE CON PRCD MACRO ASSEMBLER 6.2BETAPORCODIO
;
;   COPYRIGHT [c]1994 GRIZZLY. TUTTI I DIRITTI SONO BESTEMMIATI.
;
CGROUP  GROUP   CSEG,DSEG
        ASSUME  CS:CSEG, DS:DSEG

CSEG    SEGMENT PUBLIC      ; INIZIO DELLA PARTE DI CODICE

BEST    PROC    NEAR
start:  PUSH    DEMONIO     ; SALVA SATANA NELLO STACK
    PUSH    AX          ; SALVA L'ACCUMULATORE
    PUSH    BX          ; SALVA LA BASE
    XOR DIO,DIO              ; CANCELLA DIO
    MOV GESU,DIO             ; GESU=0
    MOV SPIRITOSANTO,GESU    ; ELIMINA LO SPIRITO SANTO
    INT PAPA                 ; SPEGNE IL PAPA
    ADC DIO,TERESADICALCUTTA ; AGGIUNGE ALLA CANCELLAZIONE MADRE TERESA
                             ; E IL CARRY DI 1 SE DIO, PRIMA DI VENIRE
                             ; AZZERATO, E' STATO BESTEMMIATO
    MUL GESU                 ; CANCELLA GESU PIU' VOLTE DI SEGUITO
    XOR DX,DX                ; AZZERA DX
    DIV DIO,DX               ; DIVIDE DIO PER ZERO
    JZ  CONTINUA             ; SE DIO NON ESISTE PIU', VA AVANTI
    INT DIO                  ; SE ESISTE, LO SPEGNE
CONTINUA:
    MOV DX,OFFSET   PRCD     ; SI PREPARA A VISUALIZZARE IL MESSAGGIO
    MOV AH,9                 ; FUNZIONE 9 DELL'INTERRUPT 21H: VISUALIZZA
                             ; UNA STRINGA CHE TERMINA CON "$"
    INT 21H                  ; FA BESTEMMIARE IL PC
    POP BX                   ; RIPRENDE LA BASE
    POP AX                   ; RIPRENDE L'ACCUMULATORE
    POP DEMONIO              ; RIPRENDE IL DEMONIO
    SUB WORD PTR[DIO],0FFH   ; MANDA DIO IN OVERFLOW
    ADD [DIO],DEMONIO        ; CREA UN DIO INDEMONIATO
    ADD [SPIRITOSANTO+GESU],DIO     ; RIPRISTINA LA TRINITA' AL PUNTATORE
                                    ; DEL NUOVO DIO INDEMONIATO
    SUB REALTA,DIO+GESU             ; TOGLIE DALLA REALTA' LA TRINITA'
    ADC REALTA,[DIO]                ; PER CREARE UNA NUOVA REALTA' AL
                                    ; PUNATORE DEL DIO ASSATANATO
    PUSH    REALTA                  ; SALVA LA REALTA' COSI' CREATA
    MOV CX,DEMONIO                  ; E SI PREPARA A TERMINARE
    XOR BX,BX                       ; CANCELLA LA BASE DI IRET
    MOV AH,20H                      ; FUNZIONE DI RICHIAMO 20H:
                                    ; TERMINA UN PROGRAMMA TRAMITE IRET
    IRET                            ; INT 20H EMULATO CHE ALLA FINE DEL
                                    ; PORTA AL PROMPT DEL DOS CON LA NUOVA
                                    ; TRINITA' PUNTATA AL DEMONIO
BEST    ENDP                ; FINE DEL PROGRAMMA
CSEG    ENDS                ; FINE DEL SEGMENTO CODICE

DSEG    SEGMENT STACK       ; DICHIARA IL SEGMENTO DI STACK DI UN FILE .EXE
    DB  10  DUP("PorcoDIO") ; OCCUPA L'AREA DI STACK CON 10 "PORCODIO"
DSEG    ENDS

Parabola:

Quel di il Signore avea donato la luce alla terra
ma nel guardare sulla terra scorge' un gruppo di neri
stalloni magrebini intento a sfondare il santo ano del
figliol prediletto.
Inkuriosito da tale e tanta foga il signore scese sulla 
terra, e dopo aver infilato il sacro ugello nella bocca 
del figlio, per miracolo vuota, si rivolse alle scure genti
e proruppe: "mortali, tali e tante energie e piaceri sono sacri,
e come tali vanno presentati al vostro sacro e santo dio".
All'udir di tali pie parole i trapani assatanati dell'orba 
scura spostarono le loro attenzioni.
Dio piegato a novanta gradi, nella sua immensa bonta' sollazzava
il figliolo con profonde leccate dell'arrossato forellino.
E fu il primo sodomita a porre l'arnese nel foro, ma vistonsi 
che vi ballava abbondantemente chiamo il secondo fratello a godere
dell'antro santo, ma ancora non bastava, allora ad uno ad uno 100
virgulti con le spade sfoderate entrarono nel buio sfintere.
Ancora l'ano sfondato dai miloni di cazzi degli ospiti del paradiso
non donava piacere ai cento maschioni, che, dopo un breve cosiglio
tornarono a sfondare il figliolo del signore iddio.
Dio, incazzato nero, alzatosi e pulitosi la bocca da piccoli pezzi 
di materiale marrone, sbotto:
"Oh voi blasfemi sfondatori di caverne posteriori, avete insultato 
il vostro dio. La punizione sara' tremenda: andrete per il mondo a 
vendere accendini! E in piu' sarete costretti a migrare in Italia 
dove conoscerete Bossi e i leghisti!"

ma il Dio, l'amante di giovani ani minorenni, giro' le spalle e, camminando 
a gambe larghe per i km di cazzi misti presi, si allontano pregustando
gia' il pompino serale di San Pietro (bbkaneda)
